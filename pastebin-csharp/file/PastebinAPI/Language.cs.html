    462         public static Language Parse(string s)
    463         {
    464             if (s == null)
    465                 throw new ArgumentNullException("s");
    466             if (TryParse(s, out Language result) == false)
    467                 throw new FormatException($"Format: {s} is not supported");
    468             return result;
    469         }
    470 
    471         public static bool TryParse(string s, out Language result)
    472         {
    473             result = Default;
    474             if (s == null || Languages.ContainsKey(s) == false)
    475                 return false;
    476             result = Languages[s];
    477             return true;
    478         }
